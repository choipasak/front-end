현재문서가 준비완료되면 수행하는 것이 jQuery
$() -> 괄호안에 위치 정해줄때 css처럼 앞에 '#'을 붙여줘야한다!
// 현재문서가 준비완료되면 수행하는 것이 jQuery
$(function() {
// 아이디가 btn1인 요소를 클릭하는 이벤트 등록
    $("#btn1").click(function(){
    //이 형식이 btn1버튼을 누르면(.click/ onclick    과 똑같음) 반응하는 익명함수
    // h2요소들의 글꼴색을 빨강으로 변경한다.
    $("h2").css("color","#ff0000");
     });
     $("#btn2").click(function(){
         $("h1").css("color","blue");
     });
});
